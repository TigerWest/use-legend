A thin wrapper around `useIntervalFn` that exposes a `ReadonlyObservable<number>` counter that increments on each tick.

## Demo

## Usage

<CodeTabs>
  <Fragment slot="hook">
    ```tsx twoslash
    // @noErrors
    import { useInterval } from "@usels/core";

    function Component() {
      const counter = useInterval(1000);
      return <div>{counter.get()}</div>;
      // counter.get() → 0 initially, then 1, 2, 3... every second
    }
    ```

  </Fragment>
  <Fragment slot="scope">
    ```tsx
    import { createInterval } from "@usels/core";

    function Component() {
      "use scope"
      const counter = createInterval(1000);
      return <div>{counter.get()}</div>;
      // counter.get() → 0 initially, then 1, 2, 3... every second
    }
    ```

  </Fragment>
</CodeTabs>

### With pause/resume/reset controls

<CodeTabs>
  <Fragment slot="hook">
    ```tsx
    import { useInterval } from "@usels/core";

    function Component() {
      const { counter$, reset, isActive$, pause, resume } = useInterval(1000, {
        controls: true,
      });

      // pause()  → stops ticking
      // resume() → restarts
      // reset()  → counter → 0
    }
    ```

  </Fragment>
  <Fragment slot="scope">
    ```tsx
    import { createInterval } from "@usels/core";

    function Component() {
      "use scope"
      const { counter$, reset, isActive$, pause, resume } = createInterval(1000, {
        controls: true,
      });

      // pause()  → stops ticking
      // resume() → restarts
      // reset()  → counter → 0
    }
    ```

  </Fragment>
</CodeTabs>

### Reactive interval

<CodeTabs>
  <Fragment slot="hook">
    ```tsx
    import { observable, useInterval } from "@usels/core";

    function Component() {
      const ms$ = observable(1000);
      const counter = useInterval(ms$);

      // ms$.set(500) → restarts with 500ms period
    }
    ```

  </Fragment>
  <Fragment slot="scope">
    ```tsx
    import { createInterval, observable } from "@usels/core";

    function Component() {
      "use scope"
      const ms$ = observable(1000);
      const counter = createInterval(ms$);

      // ms$.set(500) → restarts with 500ms period
    }
    ```

  </Fragment>
</CodeTabs>
