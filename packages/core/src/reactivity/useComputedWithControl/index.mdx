---
title: useComputedWithControl
description: "Computed Observable with explicit source control and manual trigger. Only recomputes when the declared source changes, providing fine-grained control over when expensive computations run."
category: Reactivity
type-table:
  import: "@usels/core"
  name: UseDebounced
---

## Demo

## Usage

<CodeTabs>
  <Fragment slot="hook">
    ```tsx twoslash
    // @noErrors
    import { useObservable } from "@legendapp/state/react";
    import { useComputedWithControl } from "@usels/core";

    function Component() {
      const counter$ = useObservable(1);
      const { value$, trigger } = useComputedWithControl(counter$, (count) => count * 2);
      // value$.get() === 2
      // When counter$ changes → value$ recomputes automatically
    }
    ```

  </Fragment>
  <Fragment slot="scope">
    ```tsx

    import { createComputedWithControl } from "@usels/core";

    function Component() {
      "use scope"
      const counter$ = observable(1);
      const { value$, trigger } = createComputedWithControl(counter$, (count) => count * 2);
    }
    ```

  </Fragment>
</CodeTabs>

### Manual trigger

<CodeTabs>
  <Fragment slot="hook">
    ```tsx
    import { useObservable } from "@legendapp/state/react";
    import { useComputedWithControl } from "@usels/core";

    function Component() {
      const source$ = useObservable(1);
      const { value$, trigger } = useComputedWithControl(source$, (val) => {
        return val + Math.random(); // includes non-reactive data
      });

      // trigger() forces recomputation without waiting for source change
      trigger();
    }
    ```

  </Fragment>
  <Fragment slot="scope">
    ```tsx
    import { observable } from "@legendapp/state";
    import { createComputedWithControl } from "@usels/core";

    function Component() {
      "use scope"
      const source$ = observable(1);
      const { value$, trigger } = createComputedWithControl(source$, (val) => {
        return val + Math.random();
      });

      trigger();
    }
    ```

  </Fragment>
</CodeTabs>

### Previous value (incremental computation)

<CodeTabs>
  <Fragment slot="hook">
    ```tsx
    import { useObservable } from "@legendapp/state/react";
    import { useComputedWithControl } from "@usels/core";

    function Component() {
      const event$ = useObservable(0);
      const { value$ } = useComputedWithControl(event$, (eventValue, prev) => (prev ?? 0) + eventValue);
      // Each time event$ changes, the new value is added to the accumulated total
    }
    ```

  </Fragment>
  <Fragment slot="scope">
    ```tsx
    import { observable } from "@legendapp/state";
    import { createComputedWithControl } from "@usels/core";

    function Component() {
      "use scope"
      const event$ = observable(0);
      const { value$ } = createComputedWithControl(event$, (eventValue, prev) => (prev ?? 0) + eventValue);
    }
    ```

  </Fragment>
</CodeTabs>

### Array source

<CodeTabs>
  <Fragment slot="hook">
    ```tsx

    import { useComputedWithControl } from "@usels/core";

    function Component() {
      const width$ = useObservable(100);
      const height$ = useObservable(50);
      const { value$: area$ } = useComputedWithControl([width$, height$], ([w, h]) => w * h);
      // area$.get() === 5000
      // Recomputes when either width$ or height$ changes
    }
    ```
