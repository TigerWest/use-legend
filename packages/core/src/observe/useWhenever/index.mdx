## Usage

<CodeTabs>
  <Fragment slot="hook">
    ```tsx twoslash
    // @noErrors
    import { useObservable, useWhenever } from "@usels/core";

    function Component() {
      const isReady$ = useObservable(false);

      // ✅ Effect only runs when isReady$ is truthy
      useWhenever(isReady$, (value) => {
        console.log("ready:", value);
      });
    }
    ```

  </Fragment>
  <Fragment slot="scope">
    ```tsx
    import { observable, whenever } from "@usels/core";

    function Component() {
      "use scope"
      const isReady$ = observable(false);

      // ✅ Effect only runs when isReady$ is truthy
      whenever(isReady$, (value) => {
        console.log("ready:", value);
      });
    }
    ```

  </Fragment>
</CodeTabs>

### Eager mode (`immediate: true`)

Pass `immediate: true` to also fire on setup if the value is already truthy.

<CodeTabs>
  <Fragment slot="hook">
    ```tsx
    import { useObservable, useWhenever } from "@usels/core";

    function Component() {
      const isReady$ = useObservable(true);

      useWhenever(
        isReady$,
        (value) => {
          console.log("value:", value);
        },
        { immediate: true } // fires immediately since isReady$ is truthy
      );
    }
    ```

  </Fragment>
  <Fragment slot="scope">
    ```tsx
    import { observable, whenever } from "@usels/core";

    function Component() {
      "use scope"
      const isReady$ = observable(true);

      whenever(
        isReady$,
        (value) => {
          console.log("value:", value);
        },
        { immediate: true } // fires immediately since isReady$ is truthy
      );
    }
    ```

  </Fragment>
</CodeTabs>

### Once (`once: true`)

Dispose the watcher automatically after the first truthy invocation.

<CodeTabs>
  <Fragment slot="hook">
    ```tsx
    import { useObservable, useWhenever } from "@usels/core";

    function Component() {
      const token$ = useObservable<string | null>(null);

      // ✅ Only fires the first time token$ becomes truthy
      useWhenever(
        () => token$.get(),
        (token) => {
          console.log("got token:", token);
        },
        { once: true }
      );
    }
    ```

  </Fragment>
  <Fragment slot="scope">
    ```tsx
    import { observable, whenever } from "@usels/core";

    function Component() {
      "use scope"
      const token$ = observable<string | null>(null);

      // ✅ Only fires the first time token$ becomes truthy
      whenever(
        () => token$.get(),
        (token) => {
          console.log("got token:", token);
        },
        { once: true }
      );
    }
    ```

  </Fragment>
</CodeTabs>

### Reactive function selector

Use a function to derive a truthy/falsy value from one or more observables.

<CodeTabs>
  <Fragment slot="hook">
    ```tsx
    import { useObservable, useWhenever } from "@usels/core";

    function Component() {
      const user$ = useObservable<{ name: string } | null>(null);

      useWhenever(
        () => user$.get(),
        (user) => {
          console.log("user logged in:", user.name);
        }
      );
    }
    ```

  </Fragment>
  <Fragment slot="scope">
    ```tsx
    import { observable, whenever } from "@usels/core";

    function Component() {
      "use scope"
      const user$ = observable<{ name: string } | null>(null);

      whenever(
        () => user$.get(),
        (user) => {
          console.log("user logged in:", user.name);
        }
      );
    }
    ```

  </Fragment>
</CodeTabs>

### Batch scheduling (`schedule`)

The `schedule` option controls when the effect runs relative to Legend-State's batch cycle.

- `schedule: 'sync'` — runs synchronously inside the batch
- `schedule: 'deferred'` — runs after the batch ends
- omitted — uses Legend-State's default batching

```typescript
useWhenever(flag$, (v) => console.log(v), { schedule: "sync" });
```
